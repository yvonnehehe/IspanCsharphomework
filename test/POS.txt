using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace IspanHomework
{
    public partial class POS : Form
    {
        private int totalAmount;
        private List<String> orderdetails;
        public POS()
        {
            InitializeComponent();
            totalAmount = 0;
            orderdetails= new List<String>();
        }
        private int getmenu (string menudetail)
        {
            int price = 0;
            switch (menudetail)
            {
                case "漢堡":
                    price = 140;
                    break;
                case "披薩":
                    price = 120;
                    break;
                case "薯條":
                    price = 60;
                    break;
                case "熱狗":
                    price = 135;
                    break;
            }
            return price;
        }
        private void addmenu(string menudetail)
        {
            int price = getmenu (menudetail);
            totalAmount += price;
            orderdetails.Add (menudetail);
            updatetotal();
            updateorder();

        }
        private void updatetotal()
        {
            labTotal.Text = totalAmount.ToString("C");
        }
        private void updateorder() 
        {
            listorderBox.Items.Clear();
            listorderBox.Items.AddRange(orderdetails.ToArray());
        }

        private void btnCash_Click(object sender, EventArgs e)
        {
            MessageBox.Show("總金額: " + totalAmount + "元");
        }

        private void btnHamburger_Click(object sender, EventArgs e)
        {
            addmenu("漢堡 NT$ 140");
        }

        private void btnPizza_Click(object sender, EventArgs e)
        {
            addmenu("披薩 NT$ 120");
        }

        private void btnFries_Click(object sender, EventArgs e)
        {
            addmenu("薯條 NT$ 60");
        }

        private void btnHotdog_Click(object sender, EventArgs e)
        {
            addmenu("熱狗 NT$ 135");
        }
    }
}
